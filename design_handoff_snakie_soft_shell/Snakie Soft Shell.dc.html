from snake_robot import Snake
import time

# A four-legged creep gait: lift one foot at a time so the
# centre of mass always stays over the other three.
robot = Snake("quad")

STEP = [
    ("front_left", 30),
    ("back_right", 30),
    ("front_right", 30),
    ("back_left", 30),
]

def walk(cycles=4):
    for _ in range(cycles):
        for leg, angle in STEP:
            robot.lift(leg)
            robot.swing(leg, angle)
            robot.plant(leg)
